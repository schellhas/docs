    overview

    managing multiple ssh sessions

    step 1: generating ssh keys for each account

    
ssh-keygen -t rsa -b 4096 -C "work@email.com" -f ~/.ssh/id_rsa_work
ssh-keygen -t rsa -b 4096 -C "uni@email.com" -f ~/.ssh/id_rsa_uni
ssh-keygen -t rsa -b 4096 -C "personal@email.com" -f ~/.ssh/id_rsa_personal

    step 2: add the keys to the ssh-agent

    
eval "$(ssh-agent -s)"
ssh-add ~/.ssh/id_rsa_work
ssh-add ~/.ssh/id_rsa_uni
ssh-add ~/.ssh/id_rsa_personal

    step 2.5: copy the stuff of course

    copy your the .pub key and put it in the git gui in the browser

    
cat ~/.ssh/id_rsa_work.pub
cat ~/.ssh/id_rsa_uni.pub
cat ~/.ssh/id_rsa_personal.pub

    step 3: configure the ~/.ssh/config file

    for github/gitlab, the ssh user is always git. authentication happens via the ssh key, not the username.

    example:

    
# Example for github where:
# Username: duckingduck
# URL: github.com
Host github_duckingduck			/* this is just a nickname for when you ssh */
	HostName github.com		/* actual domain or ip of the remote server */
	User git			/* your username to login the remote server */
	IdentityFile ~/.ssh/id_rsa_work	/* whatever key you generated */
	IdentitiesOnly yes		/* idk */

    longer example:

    
# work
Host work
	HostName work.com
	User git
	IdentityFile ~/.ssh/id_rsa_work
	IdentitiesOnly yes

# uni
Host uni
	HostName uni.edu
	User git
	IdentityFile ~/.ssh/id_rsa_uni
	IdentitiesOnly yes

# personal github
Host personal_github
	HostName github.com
	User git
	IdentityFile ~/.ssh/id_rsa_personal
	IdentitiesOnly yes

# personal gitlab
Host personal_gitlab
	HostName gitlab.com
	User git
	IdentityFile ~/.ssh/id_rsa_personal
	IdentitiesOnly yes

    step 4: check the connection

    
ssh -T git@github	/* for github */

    step 5: clone using ssh

    url should look something like this:

    
git clone git@github:username/repo.git

    step 6: set local user.name and user.email

    
git config user.name "name"
git config user.email "name@mail.com"
